function doPost(e) { if (!e || !e.postData || !e.postData.contents) { return ContentService .createTextOutput("Erro: requisição inválida"); } var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Avaliações"); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.chamado, data.tecnico, data.data, data.hora, data.nota, data.comentario, new Date() ]); return ContentService.createTextOutput("OK"); } function doGet(e) { const template = HtmlService.createTemplateFromFile('index'); return template.evaluate() .setTitle("Avaliação de Chamado") .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); }
